227  KeyCode translatePs2Keycode(KeyboardOutput code) {
228  KeyCode result = PS2_NONE;
229  if (code == KeyboardOutput::none) {
230  return PS2_NONE;
231  }
232  if (code == KeyboardOutput::unmake) {
233  this->isUnmake = true;
234  return PS2_NONE;
235  }
236  if (code == KeyboardOutput::extend) {
237  this->isExtended = true;
238  return PS2_NONE;
239  }
240  if (code == KeyboardOutput::extend1) {
241  this->isExtended1 = true;
242  return PS2_NONE;
243  }
244 
245  if (this->isExtended1) {
246  if (!this->haveGotExtended1FirstByte) {
247  this->haveGotExtended1FirstByte = true;
248  // Don't care about the actual content of the thing, because, oddly, there's only one
249  // key that uses Extended-1 mode
250  return PS2_NONE;
251  }
252  if (!this->isUnmake) {
253  result = KeyCode::PS2_KEY_BREAK;
254  }
255  this->reset();
256  return result;
257  }
258 
259  KeyCode modifier = this->translateModifier(code);
260  if (modifier != KeyCode::PS2_NONE) {
261  if (this->isUnmake) {
262  this->modifiers &= ~modifier;
263  }
264  else {
265  this->modifiers |= modifier;
266  }
267  this->reset();
268  return PS2_NONE;
269  }
270 
271  if (this->isUnmake) {
272  if (this->isExtended1 && !this->haveGotExtended1FirstByte) {
273  this->haveGotExtended1FirstByte = true;
274  }
275  else {
276  this->reset();
277  }
278  // Don't care about unmake
279  return PS2_NONE;
280  }
281 
282  KeyCode translatedCode = this->isExtended
283  ? translateExtended(code)
284  : translateNonExtended(code);
285  if (translatedCode != PS2_NONE) {
286  result = translatedCode | this->modifiers;
287  }
288 
289  this->reset();
290  return result;
291  }
292 
293  void reset()
294  {
295  this->isUnmake = false;
296  this->isExtended = false;
297  this->isExtended1 = false;
298  this->haveGotExtended1FirstByte = false;
299  }
